$promptSvc = @'
using System.Collections.ObjectModel;
using System.Text.Json;
using OllamaHub.Models;

namespace OllamaHub.Services;

public class PromptLibraryService
{
    private readonly string _path;
    public ObservableCollection<PromptTemplate> Templates { get; } = new();

    public PromptLibraryService()
    {
        var dir = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
            "OllamaHub");
        Directory.CreateDirectory(dir);
        _path = Path.Combine(dir, "prompts.json");
        Load();
        if (Templates.Count == 0) SeedDefaults();
    }

    private void Load()
    {
        if (!File.Exists(_path)) return;
        try
        {
            var json = File.ReadAllText(_path);
            var list = JsonSerializer.Deserialize<List<PromptTemplate>>(json);
            if (list != null) foreach (var t in list) Templates.Add(t);
        }
        catch { }
    }

    public void Save()
    {
        var json = JsonSerializer.Serialize(Templates.ToList(),
            new JsonSerializerOptions { WriteIndented = true });
        File.WriteAllText(_path, json);
    }

    public void Add(PromptTemplate t)    { Templates.Add(t); Save(); }
    public void Delete(PromptTemplate t) { Templates.Remove(t); Save(); }
    public void Update(PromptTemplate t) { Save(); }

    private void SeedDefaults()
    {
        var defaults = new[]
        {
            new PromptTemplate { Name="Explain Like I'm 5",   Category="Learning",   Content="Explain the following in simple terms:\n\n",           Icon="\uE82D" },
            new PromptTemplate { Name="Code Review",          Category="Coding",     Content="Review this code for issues:\n\n```\n\n```",          Icon="\uE943" },
            new PromptTemplate { Name="Write Unit Tests",     Category="Coding",     Content="Write tests for:\n\n```\n\n```",                       Icon="\uE8C4" },
            new PromptTemplate { Name="Summarize",            Category="Writing",    Content="Summarize:\n\n",                                      Icon="\uE8D2" },
            new PromptTemplate { Name="Fix Grammar",          Category="Writing",    Content="Fix grammar in:\n\n",                                 Icon="\uE8AB" },
            new PromptTemplate { Name="Translate",            Category="Language",   Content="Translate to English:\n\n",                          Icon="\uE8C1" },
            new PromptTemplate { Name="Debug Error",          Category="Coding",     Content="Help debug:\n\nError:\n\n```\n\n```",              Icon="\uEBE8" },
        };
        foreach (var d in defaults) Templates.Add(d);

    }
}

$ollamaApiSvc = @'
using System.Net.Http;
using System.Text;
using System.Text.Json;
using System.Threading.Tasks;

namespace OllamaHub.Services;

public class OllamaApiService
{
    private readonly HttpClient _client;
    private readonly string _baseUrl;
    private const string DefaultUrl = "http://localhost:11434";

    public OllamaApiService(string? baseUrl = null)
    {
        _baseUrl = baseUrl ?? DefaultUrl;
        _client = new HttpClient { Timeout = TimeSpan.FromMinutes(5) };
    }

    public async Task<List<string>> GetAvailableModels()
    {
        try
        {
            var response = await _client.GetAsync($"{_baseUrl}/api/tags");
            response.EnsureSuccessStatusCode();
            
            var json = await response.Content.ReadAsStringAsync(CancellationToken.None);
            using var doc = JsonDocument.Parse(json);
            
            var models = new List<string>();
            if (doc.RootElement.TryGetProperty("models", out var arr))
            {
                foreach (var model in arr.EnumerateArray())
                {
                    if (model.TryGetProperty("name", out var name))
                        models.Add(name.GetString() ?? string.Empty);
                }
            }
            return models;
        }
        catch (Exception ex)
        {
            System.Diagnostics.Debug.WriteLine($"Error fetching models: {ex.Message}");
            return new List<string>();
        }
    }

    public async IAsyncEnumerable<string> GenerateResponse(string model, string prompt)
    {
        var payload = new { model, prompt, stream = true };
        var json = JsonSerializer.Serialize(payload);
        var content = new StringContent(json, Encoding.UTF8, "application/json");

        try
        {
            var response = await _client.PostAsync($"{_baseUrl}/api/generate", content);
            response.EnsureSuccessStatusCode();

            using var stream = await response.Content.ReadAsStreamAsync(CancellationToken.None);
            using var reader = new StreamReader(stream);
            
            string? line;
            while ((line = await reader.ReadLineAsync()) != null)
            {
                if (string.IsNullOrWhiteSpace(line)) continue;
                
                try
                {
                    using var doc = JsonDocument.Parse(line);
                    if (doc.RootElement.TryGetProperty("response", out var resp))
                    {
                        yield return resp.GetString() ?? string.Empty;
                    }
                }
                catch { }
            }
        }
        catch (Exception ex)
        {
            yield return $"Error: {ex.Message}";
        }
    }
}
'@

Set-Content "$OllamaHubRoot\Services\OllamaApiService.cs" $ollamaApiSvc -Encoding UTF8
Write-Ok "Services\OllamaApiService.cs"

$studioCointSvc = @'
using System.Diagnostics;
using OllamaHub.Models;

namespace OllamaHub.Services;

public class StudioCoIntegrationService
{
    public static StudioCoIntegration GetVSInfo()
    {
        var integration = new StudioCoIntegration();
        
        try
        {
            var vsPath = FindVisualStudio();
            if (vsPath != null)
            {
                integration.IsEmbeddedInVS = true;
                integration.VSVersion = ExtractVSVersion(vsPath);
                integration.VSExtensions = GetInstalledExtensions();
            }
        }
        catch { }
        
        return integration;
    }

    private static string? FindVisualStudio()
    {
        var possiblePaths = new[]
        {
            Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.ProgramFiles), "Microsoft Visual Studio", "2022"),
            Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.ProgramFiles), "Microsoft Visual Studio", "2019"),
        };

        foreach (var basePath in possiblePaths)
        {
            if (Directory.Exists(basePath))
                return Directory.GetDirectories(basePath).FirstOrDefault();
        }
        return null;
    }

    private static string ExtractVSVersion(string path)
    {
        var parts = path.Split('\\');
        return parts.LastOrDefault() ?? "Unknown";
    }

    private static List<string> GetInstalledExtensions()
    {
        var extensionsPath = Path.Combine(
            Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData),
            "Microsoft\\VisualStudio");
        
        var extensions = new List<string>();
        if (Directory.Exists(extensionsPath))
        {
            try
            {
                var dirs = Directory.GetDirectories(extensionsPath);
                extensions.AddRange(dirs.Select(Path.GetFileName).Where(n => n != null).Cast<string>());
            }
            catch { }
        }
        return extensions;
    }
}
'@

Set-Content "$OllamaHubRoot\Services\StudioCoIntegrationService.cs" $studioCointSvc -Encoding UTF8
Write-Ok "Services\StudioCoIntegrationService.cs"

using System.Collections.ObjectModel;
using CommunityToolkit.Mvvm.ComponentModel;
using CommunityToolkit.Mvvm.Input;
using OllamaHub.Models;
using OllamaHub.Services;

namespace OllamaHub.ViewModels;

public partial class MainViewModel : ObservableObject
{
    private readonly OllamaApiService _ollama;
    private readonly PromptLibraryService _prompts;
    private readonly StudioCoIntegrationService _studioCo;

    [ObservableProperty]
    private string selectedModel = string.Empty;

    [ObservableProperty]
    private string userPrompt = string.Empty;

    [ObservableProperty]
    private string responseText = string.Empty;

    [ObservableProperty]
    private bool isLoading = false;

    [ObservableProperty]
    private ObservableCollection<string> availableModels = new();

    [ObservableProperty]
    private ObservableCollection<PromptTemplate> promptTemplates = new();

    [ObservableProperty]
    private StudioCoIntegration vsIntegration = new();

    public MainViewModel()
    {
        _ollama = new OllamaApiService();
        _prompts = new PromptLibraryService();
        _studioCo = new StudioCoIntegrationService();

        PromptTemplates = _prompts.Templates;
        VSIntegration = StudioCoIntegrationService.GetVSInfo();

        LoadModels();
    }

    [RelayCommand]
    private async Task LoadModels()
    {
        IsLoading = true;
        try
        {
            var models = await _ollama.GetAvailableModels();
            AvailableModels.Clear();
            foreach (var m in models) AvailableModels.Add(m);
            if (AvailableModels.Count > 0) SelectedModel = AvailableModels[0];
        }
        finally { IsLoading = false; }
    }

    [RelayCommand]
    private async Task GenerateResponse()
    {
        if (string.IsNullOrEmpty(SelectedModel) || string.IsNullOrEmpty(UserPrompt))
            return;

        IsLoading = true;
        ResponseText = string.Empty;
        try
        {
            await foreach (var chunk in _ollama.GenerateResponse(SelectedModel, UserPrompt))
            {
                ResponseText += chunk;
            }
        }
        finally { IsLoading = false; }
    }

    [RelayCommand]
    private void ApplyTemplate(PromptTemplate template)
    {
        UserPrompt = template.Content;
    }

    [RelayCommand]
    private void AddNewTemplate()
    {
        var newTemplate = new PromptTemplate 
        { 
            Name = "New Template",
            Content = UserPrompt 
        };
        _prompts.Add(newTemplate);
    }

    [RelayCommand]
    private void DeleteTemplate(PromptTemplate template)
    {
        _prompts.Delete(template);
    }
}
'@

using Microsoft.UI.Xaml;
using System.Diagnostics;

namespace OllamaHub;

static class Program
{

    static void Main(string[] args)
    {

        {
            Debug.WriteLine("OllamaHub v2.0 Starting...");
            WinUIApplication.Initialize(AppContext.BaseDirectory);
            var app = new App();
            app.Run();
        }
        catch (Exception ex)
        {
            Debug.WriteLine($"Fatal error: {ex}");
            throw;
        }

/// <summary>
/// WinUI Application Initialization Helper
/// </summary>
internal static class WinUIApplication
{
    private static bool _initialized = false;

    public static void Initialize(string basePath)
    {
        if (_initialized) return;
        
        // Initialize WinRT/WinUI runtime
        Windows.ApplicationModel.Core.CoreApplication.Run(
            new AppViewSource(() => new App()));
        
        _initialized = true;
    }
}

/// <summary>
/// Custom AppViewSource for WinUI3 desktop apps
/// </summary>
internal class AppViewSource : Windows.ApplicationModel.Core.IFrameworkViewSource
{
    private readonly Func<App> _appFactory;

    public AppViewSource(Func<App> appFactory)
    {
        _appFactory = appFactory;
    }

    public Windows.ApplicationModel.Core.IFrameworkView CreateView()
    {
        return new FrameworkViewWrapper(_appFactory());
    }
}

internal class FrameworkViewWrapper : Windows.ApplicationModel.Core.IFrameworkView
{
    private readonly App _app;
    private Windows.UI.Core.CoreWindow? _window;

    public FrameworkViewWrapper(App app)
    {
        _app = app;
    }

    public void Initialize(Windows.ApplicationModel.Core.CoreApplicationView applicationView)
    {
        applicationView.Activated += (s, e) =>
        {
            Windows.UI.Core.CoreWindow.GetForCurrentThread().Activate();
        };
    }

    public void SetWindow(Windows.UI.Core.CoreWindow window)
    {
        _window = window;
        _window.Closed += (s, e) =>
        {
            Windows.UI.Core.CoreApplication.Exit();
        };
    }

    public void Load(string entryPoint)
    {
    }

    public void Run()
    {
        _window?.Activate();
        Windows.UI.Core.CoreApplication.MainView.CoreWindow.Dispatcher.ProcessEvents(
            Windows.UI.Core.CoreProcessEventsOption.ProcessUntilQuit);
    }

    public void Uninitialize()
    {
    }
}
'@
